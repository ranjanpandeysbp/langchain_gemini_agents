-- ============================================================
--  MySQL DDL + DML Script
--  Tables: products, orders, order_items, sales
--  Run this script against your target database:
--    mysql -u root -p your_database < mysql_ddl_dml.sql
-- ============================================================


-- ────────────────────────────────────────────────────────────
--  SAFETY: drop tables in reverse dependency order so foreign
--  key constraints do not block the DROP statements
-- ────────────────────────────────────────────────────────────
DROP TABLE IF EXISTS sales;         -- remove sales first (references order_items)
DROP TABLE IF EXISTS order_items;   -- remove order_items next (references orders + products)
DROP TABLE IF EXISTS orders;        -- remove orders next (references customers)
DROP TABLE IF EXISTS customers;     -- remove customers next (referenced by orders)
DROP TABLE IF EXISTS products;      -- remove products last (referenced by order_items)


-- ============================================================
--  DDL — CREATE TABLES
-- ============================================================


-- ────────────────────────────────────────────────────────────
--  TABLE: products
--  Stores the product catalogue with pricing and stock info
-- ────────────────────────────────────────────────────────────
CREATE TABLE products (
    product_id      INT           AUTO_INCREMENT PRIMARY KEY,  -- unique identifier, auto-incremented by MySQL
    product_name    VARCHAR(150)  NOT NULL,                    -- human-readable product name, required
    category        VARCHAR(100)  NOT NULL,                    -- product category (e.g. Electronics, Clothing)
    unit_price      DECIMAL(10,2) NOT NULL,                    -- selling price per unit, two decimal places
    cost_price      DECIMAL(10,2) NOT NULL,                    -- purchase / manufacturing cost per unit
    stock_quantity  INT           NOT NULL DEFAULT 0,          -- current units available in warehouse
    created_at      DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP  -- row creation timestamp
);


-- ────────────────────────────────────────────────────────────
--  TABLE: customers
--  Stores customer master data referenced by orders
-- ────────────────────────────────────────────────────────────
CREATE TABLE customers (
    customer_id    INT          AUTO_INCREMENT PRIMARY KEY,  -- unique customer identifier
    first_name     VARCHAR(80)  NOT NULL,                    -- customer's first name
    last_name      VARCHAR(80)  NOT NULL,                    -- customer's last name
    email          VARCHAR(150) NOT NULL UNIQUE,             -- email address – must be unique across all customers
    phone          VARCHAR(20),                              -- optional contact phone number
    city           VARCHAR(100),                             -- customer's city for regional reporting
    created_at     DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP  -- when the customer record was created
);


-- ────────────────────────────────────────────────────────────
--  TABLE: orders
--  One row per customer order; totals are stored for quick
--  reporting without re-aggregating order_items every time
-- ────────────────────────────────────────────────────────────
CREATE TABLE orders (
    order_id        INT           AUTO_INCREMENT PRIMARY KEY,  -- unique order identifier
    customer_id     INT           NOT NULL,                    -- FK → customers.customer_id
    order_date      DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,  -- when the order was placed
    status          VARCHAR(30)   NOT NULL DEFAULT 'pending',  -- order lifecycle: pending | confirmed | shipped | delivered | cancelled
    total_amount    DECIMAL(10,2) NOT NULL DEFAULT 0.00,       -- pre-computed order total (sum of order_items)
    shipping_city   VARCHAR(100),                              -- destination city for this order
    CONSTRAINT fk_orders_customer
        FOREIGN KEY (customer_id) REFERENCES customers(customer_id)  -- enforce referential integrity
);


-- ────────────────────────────────────────────────────────────
--  TABLE: order_items
--  Line items that belong to an order; each row is one
--  product purchased within an order
-- ────────────────────────────────────────────────────────────
CREATE TABLE order_items (
    item_id       INT           AUTO_INCREMENT PRIMARY KEY,  -- unique line-item identifier
    order_id      INT           NOT NULL,                    -- FK → orders.order_id
    product_id    INT           NOT NULL,                    -- FK → products.product_id
    quantity      INT           NOT NULL,                    -- number of units purchased
    unit_price    DECIMAL(10,2) NOT NULL,                    -- price per unit at time of purchase (snapshot)
    line_total    DECIMAL(10,2) NOT NULL,                    -- quantity × unit_price, stored for fast reporting
    CONSTRAINT fk_items_order
        FOREIGN KEY (order_id)   REFERENCES orders(order_id),    -- each item must belong to a valid order
    CONSTRAINT fk_items_product
        FOREIGN KEY (product_id) REFERENCES products(product_id) -- each item must reference a valid product
);


-- ────────────────────────────────────────────────────────────
--  TABLE: sales
--  Aggregated daily sales summary per product; populated from
--  order_items for fast dashboard and analytics queries
-- ────────────────────────────────────────────────────────────
CREATE TABLE sales (
    sale_id          INT           AUTO_INCREMENT PRIMARY KEY,  -- unique sale-summary identifier
    order_item_id    INT           NOT NULL,                    -- FK → order_items.item_id (source line item)
    product_id       INT           NOT NULL,                    -- FK → products.product_id (denormalised for speed)
    sale_date        DATE          NOT NULL,                    -- calendar date of the sale (extracted from order date)
    quantity_sold    INT           NOT NULL,                    -- units sold in this transaction
    revenue          DECIMAL(10,2) NOT NULL,                    -- total income  = quantity × unit_price
    cost             DECIMAL(10,2) NOT NULL,                    -- total cost    = quantity × cost_price
    profit           DECIMAL(10,2) NOT NULL,                    -- gross profit  = revenue − cost
    CONSTRAINT fk_sales_item
        FOREIGN KEY (order_item_id) REFERENCES order_items(item_id),  -- link back to the originating line item
    CONSTRAINT fk_sales_product
        FOREIGN KEY (product_id)    REFERENCES products(product_id)   -- link to the product for easy joins
);
